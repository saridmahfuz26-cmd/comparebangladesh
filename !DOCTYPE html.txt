function formatPrice(price) {
    return '৳' + price.toLocaleString('en-BD');
}

function getStoreColor(storeName) {
    return siteColorMap[storeName] || '#a1a1aa';
}

function getStoreInfo(storeName) {
    return stores.find(s => s.name === storeName) || stores[0];
}

function performSearch() {
    const query = document.getElementById('searchInput').value.trim();
    if (!query) {
        showToast('Please enter a product name to search', 'warning');
        return;
    }
    searchProducts(query);
}

function quickSearch(query) {
    document.getElementById('searchInput').value = query;
    searchProducts(query);
}

function searchProducts(query) {
    const resultsSection = document.getElementById('resultsSection');
    const loadingState = document.getElementById('loadingState');
    const resultsContainer = document.getElementById('resultsContainer');

    resultsSection.classList.remove('hidden');
    loadingState.classList.remove('hidden');
    resultsContainer.innerHTML = '';

    resultsSection.scrollIntoView({ behavior: 'smooth', block: 'start' });

    // Simulate loading
    setTimeout(() => {
        const q = query.toLowerCase();
        currentResults = products.filter(p =>
            p.name.toLowerCase().includes(q) ||
            p.category.toLowerCase().includes(q) ||
            p.prices.some(pr => pr.store.toLowerCase().includes(q))
        );

        // If no exact match, show all with a message
        if (currentResults.length === 0) {
            currentResults = [...products].sort(() => Math.random() - 0.5).slice(0, 5);
            document.getElementById('resultsTitle').textContent = `No exact match for "${query}"`;
            document.getElementById('resultsSubtitle').textContent = 'Showing similar products you might be interested in';
        } else {
            document.getElementById('resultsTitle').textContent = `Results for "${query}"`;
            document.getElementById('resultsSubtitle').textContent = `Found ${currentResults.length} product${currentResults.length > 1 ? 's' : ''} with price comparisons`;
        }

        loadingState.classList.add('hidden');
        activeStoreFilter = 'all';
        updateStoreFilterUI();
        sortResults();
    }, 800);
}

function sortResults() {
    const sortBy = document.getElementById('sortSelect').value;
    let sorted = [...currentResults];

    switch(sortBy) {
        case 'savings':
            sorted.sort((a, b) => {
                const aMax = Math.max(...a.prices.map(p => p.price));
                const aMin = Math.min(...a.prices.map(p => p.price));
                const bMax = Math.max(...b.prices.map(p => p.price));
                const bMin = Math.min(...b.prices.map(p => p.price));
                return (bMax - bMin) - (aMax - aMin);
            });
            break;
        case 'price-low':
            sorted.sort((a, b) => Math.min(...a.prices.map(p => p.price)) - Math.min(...b.prices.map(p => p.price)));
            break;
        case 'price-high':
            sorted.sort((a, b) => Math.max(...a.prices.map(p => p.price)) - Math.max(...b.prices.map(p => p.price)));
            break;
        case 'stores':
            sorted.sort((a, b) => b.prices.length - a.prices.length);
            break;
    }

    renderResults(sorted);
}

function toggleStoreFilter(store) {
    activeStoreFilter = store;
    updateStoreFilterUI();
    sortResults();
}

function updateStoreFilterUI() {
    document.querySelectorAll('.store-filter').forEach(btn => {
        const s = btn.dataset.store;
        if (s === activeStoreFilter) {
            btn.classList.add('active');
            btn.style.borderColor = 'rgba(249,115,22,0.5)';
            btn.style.backgroundColor = 'rgba(249,115,22,0.1)';
            btn.style.color = '#fb923c';
        } else {
            btn.classList.remove('active');
            btn.style.borderColor = 'rgba(63,63,70,0.5)';
            btn.style.backgroundColor = 'transparent';
            btn.style.color = '#a1a1aa';
        }
    });
}

        `;
        grid.appendChild(card);
    });
}

function renderCategories() {
    const grid = document.getElementById('categoriesGrid');

    categories.forEach((cat, idx) => {
        const card = document.createElement('button');
        card.className = 'category-card glass-card rounded-xl p-4 text-center hover:border-orange-500/30 transition-all duration-300 animate-in';
        card.style.animationDelay = `${idx * 0.05}s`;
        card.style.opacity = '0';
        card.onclick = () => {
            if (cat.count > 0) {
                quickSearch(cat.name);
            } else {
                showToast(`${cat.name} products coming soon!`, 'info');
            }
        };

        card.innerHTML = `
            <div class="cat-icon w-10 h-10 mx-auto rounded-xl bg-zinc-800/50 flex items-center justify-center mb-2.5">
                <span class="iconify text-xl text-zinc-400" data-icon="${cat.icon}"></span>
            </div>
            <div class="text-sm font-medium text-zinc-300">${cat.name}</div>
            <div class="text-[10px] text-zinc-600 mt-0.5">${cat.count > 0 ? cat.count + ' products' : 'Coming soon'}</div>
        `;
        grid.appendChild(card);
    });
}
